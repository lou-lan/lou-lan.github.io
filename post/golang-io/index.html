// golang.org/x/sys/unix/ztypes_darwin_amd64.go:282
type Kevent_t struct {
	Ident  uint64  // 事件标识
	Filter int16   // 监听事件的类型
	Flags  uint16  // 事件操作类型
	Fflags uint32  // 特定 filter 的专有标志，可用于保存专有返回信息
	Data   int64   // 特定 filter 存储专有信息
	Udata  *byte   // 可携带的任意用户数据
}
